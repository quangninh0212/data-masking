// ================== SHA-256 ==================
void sha256Hash(const char* input, char* output) {
  unsigned char hash[32];
  mbedtls_sha256_context ctx;
  mbedtls_sha256_init(&ctx);
  mbedtls_sha256_starts(&ctx, 0);
  mbedtls_sha256_update(&ctx, (const unsigned char*)input, strlen(input));
  mbedtls_sha256_finish(&ctx, hash);
  mbedtls_sha256_free(&ctx);
  for (int i = 0; i < 32; i++) sprintf(output + (i * 2), "%02x", hash[i]);
  output[64] = '\0';
}

// ================== NVS ==================
void loadCredentials() {
  if (xSemaphoreTake(xMutexPrefs, pdMS_TO_TICKS(2000)) == pdTRUE) {
    preferences.begin("doorauth", true);
    String pin = preferences.getString("pinHash", "");
    if (pin.length() > 0) pin.toCharArray(storedPinHash, 65);
    else sha256Hash("1234", storedPinHash);

    for (int i = 0; i < MAX_RFID_CARDS; i++) {
      String key = "rfid" + String(i);
      String hash = preferences.getString((key + "_hash").c_str(), "");
      String name = preferences.getString((key + "_name").c_str(), "");
      if (hash.length() > 0) {
        hash.toCharArray(rfidWhitelist[i].uidHash, 65);
        name.toCharArray(rfidWhitelist[i].username, 32);
        rfidWhitelist[i].active = true;
      } else {
        rfidWhitelist[i].active = false;
      }
    }
    preferences.end();
    xSemaphoreGive(xMutexPrefs);
  }
  Serial.println("✅ Credentials loaded from NVS");
}

void saveCredentials() {
  if (xSemaphoreTake(xMutexPrefs, pdMS_TO_TICKS(2000)) == pdTRUE) {
    preferences.begin("doorauth", false);
    preferences.putString("pinHash", storedPinHash);
    for (int i = 0; i < MAX_RFID_CARDS; i++) {
      String key = "rfid" + String(i);
      if (rfidWhitelist[i].active) {
        preferences.putString((key + "_hash").c_str(), rfidWhitelist[i].uidHash);
        preferences.putString((key + "_name").c_str(), rfidWhitelist[i].username);
      } else {
        preferences.remove((key + "_hash").c_str());
        preferences.remove((key + "_name").c_str());
      }
    }
    preferences.end();
    xSemaphoreGive(xMutexPrefs);
  }
  Serial.println("✅ Credentials saved to NVS");
}

// ================== Auth Helpers ==================
bool validateRfidHash(String uid) {
  char uidHash[65];
  sha256Hash(uid.c_str(), uidHash);
  for (int i = 0; i < MAX_RFID_CARDS; i++) {
    if (rfidWhitelist[i].active && strcmp(uidHash, rfidWhitelist[i].uidHash) == 0) {
      Serial.printf("✅ RFID valid: %s\n", rfidWhitelist[i].username);
      return true;
    }
  }
  return false;
}

bool validatePinHash(String pin) {
  char pinHash[65];
  sha256Hash(pin.c_str(), pinHash);
  return strcmp(pinHash, storedPinHash) == 0;
}

// ================== MQTT Queue Publish (thread-safe) ==================
void mqttPublishQueued(const char* topic, const char* payload) {
  MqttMsg msg;
  strncpy(msg.topic, topic, sizeof(msg.topic) - 1);
  strncpy(msg.payload, payload, sizeof(msg.payload) - 1);
  msg.topic[sizeof(msg.topic) - 1] = '\0';
  msg.payload[sizeof(msg.payload) - 1] = '\0';
  xQueueSend(xQueueMqttOut, &msg, pdMS_TO_TICKS(100));
}

void mqttPublishJson(const char* topic, JsonDocument& doc) {
  char buffer[256];
  serializeJson(doc, buffer);
  mqttPublishQueued(topic, buffer);
}

// ================== Hardware Helpers ==================
void buzzerBeep(int times, int duration, int pause) {
  for (int i = 0; i < times; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    vTaskDelay(pdMS_TO_TICKS(duration));
    digitalWrite(BUZZER_PIN, LOW);
    if (i < times - 1) vTaskDelay(pdMS_TO_TICKS(pause));
  }
}

void moveServoSmooth(Servo& servo, int startAngle, int endAngle, int stepDelay) {
  if (startAngle < endAngle) {
    for (int pos = startAngle; pos <= endAngle; pos++) {
      servo.write(pos);
      vTaskDelay(pdMS_TO_TICKS(stepDelay));
    }
  } else {
    for (int pos = startAngle; pos >= endAngle; pos--) {
      servo.write(pos);
      vTaskDelay(pdMS_TO_TICKS(stepDelay));
    }
  }
}

void neoPixelOff() {
  for (int i = 0; i < NUMPIXELS; i++) pixels.setPixelColor(i, 0);
  pixels.show();
}

void setBedColor(int index) {
  bedColorIndex = index;
  if (index == 0) {
    neoPixelOff();
    return;
  }
  uint32_t color;
  switch (index) {
    case 1: color = pixels.Color(255, 0, 0); break;
    case 2: color = pixels.Color(0, 255, 0); break;
    case 3: color = pixels.Color(0, 0, 255); break;
    case 4: color = pixels.Color(255, 255, 0); break;
    case 5: color = pixels.Color(255, 0, 255); break;
    case 6: color = pixels.Color(0, 255, 255); break;
    case 7: color = pixels.Color(255, 255, 255); break;
    default: color = 0;
  }
  for (int i = 0; i < NUMPIXELS; i++) pixels.setPixelColor(i, color);
  pixels.show();
}

void setNeoPixelColor(const char* hexColor) {
  neoPixelColorHex = String(hexColor);
  long colorValue = strtol(hexColor + 1, NULL, 16);
  uint8_t r = (colorValue >> 16) & 0xFF;
  uint8_t g = (colorValue >> 8) & 0xFF;
  uint8_t b = colorValue & 0xFF;
  for (int i = 0; i < NUMPIXELS; i++) pixels.setPixelColor(i, pixels.Color(r, g, b));
  pixels.show();
  bedColorIndex = 7;
}

// ================== Door Alarm Helpers ==================
void triggerDoorAlarm() {
  doorAlarmActive = true;
  Serial.println("🚨 DOOR ALARM - 5 failed attempts!");
  publishDoorAlarm("max_failed_attempts");
  digitalWrite(LED_ALARM, HIGH);
  for (int i = 0; i < 10; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    vTaskDelay(pdMS_TO_TICKS(200));
    digitalWrite(BUZZER_PIN, LOW);
    vTaskDelay(pdMS_TO_TICKS(100));

  if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {

    lcd01.setCursor(0, 1);

  }
}

void resetDoorAlarm() {
  doorAlarmActive = false;
  failedAttempts = 0;
  digitalWrite(LED_ALARM, LOW);
  digitalWrite(BUZZER_PIN, LOW);
  if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
    lcd01.clear();
    lcd01.print("Alarm reset");
    xSemaphoreGive(xMutexLcd01);

  vTaskDelay(pdMS_TO_TICKS(1000));
  if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
    lcd01.clear();
    lcd01.print("Nhap mat khau:");
    xSemaphoreGive(xMutexLcd01);
  }
  publishDoorStatusEnhanced();
  Serial.println("✅ Door alarm reset");
}

void handleFailedDoorAttempt(String reason, String rfidUid) {
  failedAttempts++;
  Serial.printf("❌ Failed attempt %d/5: %s\n", failedAttempts, reason.c_str());
  buzzerBeep(1, 500, 0);
  digitalWrite(LED_ALARM, HIGH);
  vTaskDelay(pdMS_TO_TICKS(300));
  digitalWrite(LED_ALARM, LOW);
  publishDoorAccessLog("access_denied", rfidUid, reason);
  if (failedAttempts >= MAX_FAILED_ATTEMPTS) triggerDoorAlarm();
  rfidValidated = false;
  currentRfidUid = "";
  inputPin = "";
}

// ================== MQTT Callback (runs in MQTT task context) ==================
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  if (length >= 512) return;  // safety
  MqttCmd cmd;
  strncpy(cmd.topic, topic, sizeof(cmd.topic) - 1);
  cmd.topic[sizeof(cmd.topic) - 1] = '\0';
  memcpy(cmd.payload, payload, length);
  cmd.payload[length] = '\0';
  xQueueSend(xQueueMqttIn, &cmd, pdMS_TO_TICKS(50));
}

// ================== Process one inbound MQTT command ==================
void processMqttCommand(const char* topic, const char* message) {
  Serial.printf("📥 MQTT [%s]: %s\n", topic, message);

  StaticJsonDocument<512> doc;
  if (deserializeJson(doc, message) != DeserializationError::Ok) {
    Serial.println("❌ JSON parse failed");
    return;
  }

  String topicStr = String(topic);
  String action = doc["action"] | "";
  String device = doc["device"] | "";

  // ---- Fan ----
  if (topicStr == TOPIC_FAN_CTRL) {
    fanState = (action == "on");
    digitalWrite(RELAY_FAN, fanState ? LOW : HIGH);
    if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
      lcd02.clear();
      lcd02.print(fanState ? "Quat: ON" : "Quat: OFF");
      xSemaphoreGive(xMutexLcd02);
    }
    publishFanState();
  }
  // ---- Pump ----
  else if (topicStr == TOPIC_PUMP_CTRL) {
    pumpState = (action == "on");
    digitalWrite(RELAY_PUMP, pumpState ? LOW : HIGH);
    if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
      lcd02.clear();
      lcd02.print(pumpState ? "Bom: ON" : "Bom: OFF");
      xSemaphoreGive(xMutexLcd02);
    }
    publishPumpState();
  }
  // ---- Door ----
  else if (topicStr == TOPIC_DOOR_CTRL) {
    if (action == "open" && !fireDoorOpened) {
      buzzerBeep(2, 100, 100);
      openDoorSmooth();
      fireDoorOpened = true;
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print("Cua: MO (MQTT)");
        xSemaphoreGive(xMutexLcd02);
      }
      publishDoorAccessLog("access_granted", "MQTT", "web_admin");
      failedAttempts = 0;
    } else if (action == "close" && fireDoorOpened) {
      buzzerBeep(2, 100, 100);
      closeDoorSmooth();
      fireDoorOpened = false;
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print("Cua: DONG (MQTT)");
        xSemaphoreGive(xMutexLcd02);
      }
    }
    publishDoorState();
  }
  // ---- Light ----
  else if (topicStr == TOPIC_LIGHT_CTRL) {
    if (device == "living") {
      lightLiv = (action == "on");
      digitalWrite(LIGHT_LIV, lightLiv ? HIGH : LOW);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print(lightLiv ? "Den khach: ON" : "Den khach: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
    } else if (device == "outside") {
      if (action == "auto") {
        lightOutAuto = true;
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Den ngoai: AUTO");
          xSemaphoreGive(xMutexLcd02);
        }
      } else {
        lightOutAuto = false;
        digitalWrite(LIGHT_OUT, LOW);
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Den ngoai: OFF");
          xSemaphoreGive(xMutexLcd02);
        }
      }
    } else if (device == "bedroom") {
      lightBed = (action == "on");
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print(lightBed ? "Den ngu: ON" : "Den ngu: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
    } else if (device == "neopixel") {
      if (action == "on") {
        const char* colorHex = doc["color"] | "#FFFFFF";
        setNeoPixelColor(colorHex);
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("NeoPixel: ON");
          xSemaphoreGive(xMutexLcd02);
        }
      } else {
        setBedColor(0);
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("NeoPixel: OFF");
          xSemaphoreGive(xMutexLcd02);
        }
      }
    }
    publishLightState();
  }
  // ---- Alert ----
  else if (topicStr == TOPIC_ALERT_CTRL) {
    if (action == "on") {
      emergencyMode = true;
      alarmMuted = false;
      digitalWrite(BUZZER_PIN, HIGH);
      digitalWrite(LED_ALARM, HIGH);
      digitalWrite(RELAY_FAN, LOW);
      digitalWrite(RELAY_PUMP, LOW);
      fanState = true;
      pumpState = true;
      if (!fireDoorOpened) {
        openDoorSmooth();
        fireDoorOpened = true;
      }
      lightOutAuto = false;
      lightLiv = false;
      digitalWrite(LIGHT_OUT, LOW);
      digitalWrite(LIGHT_LIV, LOW);
      setBedColor(0);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print("CANH BAO: ON");
        xSemaphoreGive(xMutexLcd02);
      }
    } else if (action == "off") {
      emergencyMode = false;
      alarmMuted = false;
      digitalWrite(BUZZER_PIN, LOW);
      digitalWrite(LED_ALARM, LOW);
      digitalWrite(RELAY_FAN, HIGH);
      digitalWrite(RELAY_PUMP, HIGH);
      fanState = false;
      pumpState = false;
      if (fireDoorOpened) {
        closeDoorSmooth();
        fireDoorOpened = false;
      }
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print("CANH BAO: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
      if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd01.clear();
        lcd01.print("Nhap mat khau:");
        xSemaphoreGive(xMutexLcd01);
      }
    } else if (action == "mute") {
      alarmMuted = true;
      digitalWrite(BUZZER_PIN, LOW);
    }
    publishAllStates();
  }
  // ---- Door Config PIN ----
  else if (topicStr == TOPIC_DOOR_CONFIG_PIN) {
    const char* newPinHash = doc["pinHash"];
    if (newPinHash && strlen(newPinHash) == 64) {
      strcpy(storedPinHash, newPinHash);
      saveCredentials();
      Serial.println("✅ PIN updated via MQTT");
    }
  }
  // ---- Door Config RFID Whitelist ----
  else if (topicStr == TOPIC_DOOR_CONFIG_RFID) {
    JsonArray whitelist = doc["whitelist"];
    for (int i = 0; i < MAX_RFID_CARDS; i++) rfidWhitelist[i].active = false;
    int idx = 0;
    for (JsonObject card : whitelist) {
      if (idx >= MAX_RFID_CARDS) break;
      const char* uidHash = card["uidHash"];
      const char* username = card["username"] | "User";
      if (uidHash && strlen(uidHash) == 64) {
        strcpy(rfidWhitelist[idx].uidHash, uidHash);
        strncpy(rfidWhitelist[idx].username, username, 31);
        rfidWhitelist[idx].active = true;
        idx++;
      }
    }
    saveCredentials();
    Serial.printf("✅ RFID whitelist updated: %d cards\n", idx);
  }
  // ---- Door Command (remote unlock / reset alarm) ----
  else if (topicStr == TOPIC_DOOR_COMMAND) {
    if (action == "unlock") {
      buzzerBeep(2, 100, 100);
      openAndCloseDoorSmooth();
      publishDoorAccessLog("access_granted", "REMOTE", "web_admin");
      failedAttempts = 0;
    } else if (action == "reset_alarm") {
      resetDoorAlarm();
    }
  }
  // ---- RFID Enrollment Start/Cancel ----
  else if (topicStr == TOPIC_DOOR_ENROLLMENT) {
    if (action == "start") {
      enrollmentMode = true;
      enrollmentStartTime = millis();
      const char* uname = doc["username"] | "User";
      enrollmentUsername = String(uname);
      if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd01.clear();
        lcd01.print("Dang ky the:");
        lcd01.setCursor(0, 1);
        lcd01.print(enrollmentUsername.substring(0, 16));
        xSemaphoreGive(xMutexLcd01);
      }
      buzzerBeep(2, 100, 100);
      Serial.printf("📝 Enrollment started for: %s\n", uname);
    } else if (action == "cancel") {
      enrollmentMode = false;
      enrollmentUsername = "";
      if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd01.clear();
        lcd01.print("Huy dang ky");
        xSemaphoreGive(xMutexLcd01);
      }
      vTaskDelay(pdMS_TO_TICKS(1000));
      if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd01.clear();
        lcd01.print("Nhap mat khau:");
        xSemaphoreGive(xMutexLcd01);
      }
      buzzerBeep(1, 500, 0);
    }
  }
  // ---- Enrollment Result from server ----
  else if (topicStr == TOPIC_DOOR_ENROLLMENT_RESULT) {
    bool success = doc["success"] | false;
    const char* resultMsg = doc["message"] | "";
    enrollmentMode = false;
    enrollmentUsername = "";
    if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
      lcd01.clear();
      if (success) {
        lcd01.print("Dang ky OK!");
        lcd01.setCursor(0, 1);
        lcd01.print(String(resultMsg).substring(0, 16));
        buzzerBeep(3, 100, 100);
      } else {
        lcd01.print("Dang ky FAIL!");
        lcd01.setCursor(0, 1);
        lcd01.print(String(resultMsg).substring(0, 16));
        buzzerBeep(1, 1000, 0);
      }
      xSemaphoreGive(xMutexLcd01);
    }
    vTaskDelay(pdMS_TO_TICKS(3000));
    if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
      lcd01.clear();
      lcd01.print("Nhap mat khau:");
      xSemaphoreGive(xMutexLcd01);
    }
  }
}

// ---- Task 1: MQTT (Core 0) --------------------------------------------------
// Owns WiFiClientSecure + PubSubClient loop, reconnect, drains publish queue,
// dispatches inbound command queue to processMqttCommand().
void TaskMQTT(void* pvParameters) {
  unsigned long lastReconnect = 0;
  const unsigned long RECONNECT_INTERVAL = 5000;

  for (;;) {
    // Reconnect WiFi if dropped
    if (WiFi.status() != WL_CONNECTED) {
      wifiConnected = false;
      mqttConnected = false;
      Serial.println("📶 WiFi lost – reconnecting...");
      WiFi.reconnect();
      vTaskDelay(pdMS_TO_TICKS(5000));
      continue;
    }
    wifiConnected = true;

    // Reconnect MQTT if needed
    if (!mqtt.connected()) {
      mqttConnected = false;
      if (millis() - lastReconnect > RECONNECT_INTERVAL) {
        lastReconnect = millis();
        Serial.println("🔗 Connecting MQTT...");
        String clientId = "esp32_tls_" + String(random(0xffff), HEX);
        if (mqtt.connect(clientId.c_str(), mqtt_user, mqtt_pass_str)) {
          Serial.println("✅ MQTT connected!");
          mqttConnected = true;
          if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd02.clear();
            lcd02.print("MQTT: OK");
            xSemaphoreGive(xMutexLcd02);
          }
          mqtt.subscribe(TOPIC_FAN_CTRL);
          mqtt.subscribe(TOPIC_PUMP_CTRL);
          mqtt.subscribe(TOPIC_DOOR_CTRL);
          mqtt.subscribe(TOPIC_LIGHT_CTRL);
          mqtt.subscribe(TOPIC_ALERT_CTRL);
          mqtt.subscribe(TOPIC_DOOR_CONFIG_PIN);
          mqtt.subscribe(TOPIC_DOOR_CONFIG_RFID);
          mqtt.subscribe(TOPIC_DOOR_COMMAND);
          mqtt.subscribe(TOPIC_DOOR_ENROLLMENT);
          mqtt.subscribe(TOPIC_DOOR_ENROLLMENT_RESULT);
          publishAllStates();
          publishDoorStatusEnhanced();
        } else {
          Serial.printf("❌ MQTT failed, rc=%d\n", mqtt.state());
          if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd02.clear();
            lcd02.print("MQTT: Offline");
            xSemaphoreGive(xMutexLcd02);
          }
          vTaskDelay(pdMS_TO_TICKS(RECONNECT_INTERVAL));
          continue;
        }
      }
    } else {
      mqttConnected = true;
    }

    // Drain outgoing publish queue
    MqttMsg outMsg;
    while (xQueueReceive(xQueueMqttOut, &outMsg, 0) == pdTRUE) {
      if (xSemaphoreTake(xMutexMqtt, pdMS_TO_TICKS(500)) == pdTRUE) {
        mqtt.publish(outMsg.topic, outMsg.payload);
        xSemaphoreGive(xMutexMqtt);
      }
    }

    // Process inbound command queue
    MqttCmd inCmd;
    while (xQueueReceive(xQueueMqttIn, &inCmd, 0) == pdTRUE) {
      processMqttCommand(inCmd.topic, inCmd.payload);
    }

    // Heartbeat
    if (millis() - lastHeartbeat > HEARTBEAT_INTERVAL) {
      lastHeartbeat = millis();
      StaticJsonDocument<64> doc;
      doc["deviceId"] = device_id;
      doc["uptime"] = millis() / 1000;
      mqttPublishJson(TOPIC_HEARTBEAT, doc);
    }

    // Periodic door status
    if (millis() - lastDoorStatus > DOOR_STATUS_INTERVAL) {
      lastDoorStatus = millis();
      publishDoorStatusEnhanced();
    }

    // Must call mqtt.loop() regularly
    if (xSemaphoreTake(xMutexMqtt, pdMS_TO_TICKS(100)) == pdTRUE) {
      mqtt.loop();
      xSemaphoreGive(xMutexMqtt);
    }

    vTaskDelay(pdMS_TO_TICKS(10));
  }
}

// ---- Task 2: Sensors (Core 0) -----------------------------------------------
// Reads gas / flame / LDR / rain, handles fire+gas emergency, publishes sensors.
void TaskSensors(void* pvParameters) {
  const int FLAME_THRESHOLD = 500;
  const int GAS_THRESHOLD = 2000;
  const unsigned long PUB_INTERVAL = 500;

  unsigned long lastPublish = 0;
  unsigned long dangerEndTime = 0;
  bool dangerActive = false;

  for (;;) {
    int flameValue = analogRead(FLAME_SENSOR);
    int gasValue = analogRead(GAS_SENSOR);

    // ---- Emergency Detection ----
    if (!alarmMuted && (flameValue < FLAME_THRESHOLD || gasValue > GAS_THRESHOLD)) {
      dangerActive = true;
      dangerEndTime = millis() + 5000;

      digitalWrite(BUZZER_PIN, HIGH);
      digitalWrite(LED_ALARM, HIGH);

      if (flameValue < FLAME_THRESHOLD) {
        emergencyMode = true;
        fanState = true;
        pumpState = true;
        digitalWrite(RELAY_FAN, LOW);
        digitalWrite(RELAY_PUMP, LOW);
        lightOutAuto = false;
        lightLiv = false;
        digitalWrite(LIGHT_OUT, LOW);
        digitalWrite(LIGHT_LIV, LOW);
        neoPixelOff();
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(200)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("LUA! CANH BAO");
          xSemaphoreGive(xMutexLcd01);
        }
        if (!fireDoorOpened) {
          openDoorSmooth();
          fireDoorOpened = true;
        }
        if (!fireAlertSent) {
          StaticJsonDocument<200> doc;
          doc["type"] = "fire";
          doc["level"] = "CRITICAL";
          doc["message"] = "CANH BAO CHAY! Phat hien lua!";
          doc["value"] = flameValue;
          mqttPublishJson("home/alert", doc);
          fireAlertSent = true;
        }
        publishAllStates();
      } else if (gasValue > GAS_THRESHOLD) {
        fanState = true;
        pumpState = false;
        digitalWrite(RELAY_FAN, LOW);
        digitalWrite(RELAY_PUMP, HIGH);
        lightOutAuto = false;
        lightLiv = false;
        digitalWrite(LIGHT_OUT, LOW);
        digitalWrite(LIGHT_LIV, LOW);
        neoPixelOff();
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(200)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("GAS! CANH BAO");
          xSemaphoreGive(xMutexLcd01);
        }
        if (!fireDoorOpened) {
          openDoorSmooth();
          fireDoorOpened = true;
        }
        if (!gasAlertSent) {
          StaticJsonDocument<200> doc;
          doc["type"] = "gas";
          doc["level"] = "CRITICAL";
          doc["message"] = "CANH BAO GAS! Nong do vuot nguong!";
          doc["value"] = gasValue;
          mqttPublishJson("home/alert", doc);
          gasAlertSent = true;
        }
        publishAllStates();
      }
    } else if (!alarmMuted && dangerActive) {
      if (millis() < dangerEndTime) {
        digitalWrite(BUZZER_PIN, HIGH);
        digitalWrite(LED_ALARM, HIGH);
      } else {
        dangerActive = false;
        emergencyMode = false;
        digitalWrite(BUZZER_PIN, LOW);
        digitalWrite(LED_ALARM, LOW);
        if (fireDoorOpened) {
          closeDoorSmooth();
          fireDoorOpened = false;
        }
        fanState = false;
        pumpState = false;
        digitalWrite(RELAY_FAN, HIGH);
        digitalWrite(RELAY_PUMP, HIGH);
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(200)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Nhap mat khau:");
          xSemaphoreGive(xMutexLcd01);
        }
        fireAlertSent = false;
        gasAlertSent = false;
        publishAllStates();
      }
    }

    // ---- Auto Outdoor Light ----
    if (lightOutAuto) {
      int lightState = digitalRead(LDR_SENSOR);
      digitalWrite(LIGHT_OUT, lightState == LOW ? LOW : HIGH);
    }

    // ---- Publish sensor data ----
    if (millis() - lastPublish > PUB_INTERVAL) {
      lastPublish = millis();
      if (mqttConnected) {
        int lightValue = !digitalRead(LDR_SENSOR);
        int rainValue = !digitalRead(RAIN_SENSOR);

        StaticJsonDocument<128> doc;
        doc["level"] = gasValue;
        doc["threshold"] = GAS_THRESHOLD;
        mqttPublishJson(TOPIC_SENSOR_GAS, doc);

        doc.clear();
        doc["detected"] = (flameValue < FLAME_THRESHOLD);
        doc["value"] = flameValue;
        doc["location"] = "main_room";
        mqttPublishJson(TOPIC_SENSOR_FIRE, doc);

        doc.clear();
        doc["bright"] = (lightValue == 1);
        mqttPublishJson(TOPIC_SENSOR_LIGHT, doc);

        doc.clear();
        doc["raining"] = (rainValue == 1);
        mqttPublishJson(TOPIC_SENSOR_RAIN, doc);
      }
    }

    vTaskDelay(pdMS_TO_TICKS(100));
  }
}

// ---- Task 3: Keypad + RFID (Core 1) -----------------------------------------
void TaskKeypadRFID(void* pvParameters) {
  for (;;) {
    // Skip all door logic when door alarm is active (handled in TaskAlarm)
    if (doorAlarmActive) {
      vTaskDelay(pdMS_TO_TICKS(200));
      continue;
    }

    // ---- RFID Enrollment Mode ----
    if (enrollmentMode) {
      if (millis() - enrollmentStartTime > ENROLLMENT_TIMEOUT_MS) {
        enrollmentMode = false;
        enrollmentUsername = "";
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Het thoi gian!");
          xSemaphoreGive(xMutexLcd01);
        }
        vTaskDelay(pdMS_TO_TICKS(2000));
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Nhap mat khau:");
          xSemaphoreGive(xMutexLcd01);
        }
        Serial.println("⏰ Enrollment timeout");
        vTaskDelay(pdMS_TO_TICKS(100));
        continue;
      }

      // Blink LED during enrollment
      digitalWrite(LED_ALARM, (millis() / 300) % 2);

      if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
        String scannedUid = "";
        for (byte i = 0; i < mfrc522.uid.size; i++)
          scannedUid += String(mfrc522.uid.uidByte[i], HEX);
        scannedUid.toUpperCase();

        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Dang xu ly...");
          xSemaphoreGive(xMutexLcd01);
        }

        if (mqttConnected) {
          StaticJsonDocument<128> doc;
          doc["uid"] = scannedUid;
          doc["username"] = enrollmentUsername;
          doc["timestamp"] = millis();
          mqttPublishJson(TOPIC_DOOR_ENROLLMENT_SCAN, doc);
        }
        buzzerBeep(1, 200, 0);
        mfrc522.PICC_HaltA();
        mfrc522.PCD_StopCrypto1();
      }

      // Cancel enrollment from keypad
      char key = keypad.getKey();
      if (key == '*') {
        enrollmentMode = false;
        enrollmentUsername = "";
        digitalWrite(LED_ALARM, LOW);
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Huy dang ky");
          xSemaphoreGive(xMutexLcd01);
        }
        vTaskDelay(pdMS_TO_TICKS(1000));
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Nhap mat khau:");
          xSemaphoreGive(xMutexLcd01);
        }
        buzzerBeep(1, 500, 0);
      }
      vTaskDelay(pdMS_TO_TICKS(50));
      continue;
    }

    // ---- Normal LED off when not in alarm ----
    digitalWrite(LED_ALARM, LOW);

    // ---- RFID Scan (hash-based) ----
    if (!rfidValidated) {
      if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
        currentRfidUid = "";
        for (byte i = 0; i < mfrc522.uid.size; i++)
          currentRfidUid += String(mfrc522.uid.uidByte[i], HEX);
        currentRfidUid.toUpperCase();
        Serial.printf("🏷️ RFID: %s\n", currentRfidUid.c_str());

        bool validHash = validateRfidHash(currentRfidUid);

        // Legacy UID fallback
        bool validLegacy = false;
        byte validUID[4] = { 0x3A, 0x4D, 0x32, 0x16 };
        if (mfrc522.uid.size == 4) {
          validLegacy = true;
          for (byte i = 0; i < 4; i++) {
            if (mfrc522.uid.uidByte[i] != validUID[i]) {
              validLegacy = false;
              break;
            }
          }
        }

        if (validHash) {
          rfidValidated = true;
          rfidTimeout = millis();
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print("Nhap PIN:");
            xSemaphoreGive(xMutexLcd01);
          }
          buzzerBeep(1, 100, 0);
        } else if (validLegacy) {
          buzzerBeep(2, 100, 100);
          openAndCloseDoorSmooth();
          publishDoorAccessLog("access_granted", currentRfidUid, "rfid_legacy");
          failedAttempts = 0;
        } else {
          handleFailedDoorAttempt("invalid_rfid", currentRfidUid);
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print("The sai!");
            xSemaphoreGive(xMutexLcd01);
          }
          vTaskDelay(pdMS_TO_TICKS(2000));
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print("Nhap mat khau:");
            xSemaphoreGive(xMutexLcd01);
          }
        }
        mfrc522.PICC_HaltA();
        mfrc522.PCD_StopCrypto1();
      }
    }

    // ---- PIN entry after RFID ----
    if (rfidValidated) {
      if (millis() - rfidTimeout > RFID_TIMEOUT_MS) {
        handleFailedDoorAttempt("pin_timeout", currentRfidUid);
        if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd01.clear();
          lcd01.print("Nhap mat khau:");
          xSemaphoreGive(xMutexLcd01);
        }
        vTaskDelay(pdMS_TO_TICKS(50));
        continue;
      }

      char key = keypad.getKey();
      if (key) {
        buzzerBeep(1, 50, 0);
        if (key == '#') {
          if ((int)inputPin.length() == PIN_LENGTH) {
            if (validatePinHash(inputPin)) {
              buzzerBeep(2, 100, 100);
              openAndCloseDoorSmooth();
              publishDoorAccessLog("access_granted", currentRfidUid, "rfid_pin");
              failedAttempts = 0;
              rfidValidated = false;
              currentRfidUid = "";
              inputPin = "";
            } else {
              handleFailedDoorAttempt("invalid_pin", currentRfidUid);
              if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
                lcd01.clear();
                lcd01.print("Nhap mat khau:");
                xSemaphoreGive(xMutexLcd01);
              }
            }
          } else {
            handleFailedDoorAttempt("invalid_pin_length", currentRfidUid);
            if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
              lcd01.clear();
              lcd01.print("Nhap mat khau:");
              xSemaphoreGive(xMutexLcd01);
            }
          }
        } else if (key == '*') {
          inputPin = "";
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.setCursor(0, 1);
            lcd01.print("                ");
            xSemaphoreGive(xMutexLcd01);
          }
        } else if (key >= '0' && key <= '9') {
          if ((int)inputPin.length() < PIN_LENGTH) {
            inputPin += key;
            if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
              lcd01.setCursor(0, 1);
              for (unsigned int i = 0; i < inputPin.length(); i++) lcd01.print("*");
              xSemaphoreGive(xMutexLcd01);
            }
          }
        }
      }
      vTaskDelay(pdMS_TO_TICKS(50));
      continue;
    }

    // ---- Keypad: normal password entry ----
    char key = keypad.getKey();
    if (key) {
      buzzerBeep(1, 100, 0);

      if (!changePassMode) {
        if (key == '#') {
          bool validHash = validatePinHash(inputPass);
          bool validLegacy = (inputPass == passwordSet);
          if (validHash || validLegacy) {
            buzzerBeep(2, 100, 100);
            openAndCloseDoorSmooth();
            publishDoorAccessLog("access_granted", "KEYPAD", validHash ? "pin_hash" : "pin_legacy");
            failedAttempts = 0;
          } else {
            handleFailedDoorAttempt("invalid_pin", "KEYPAD");
            if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
              lcd01.clear();
              lcd01.print("Sai mat khau!");
              xSemaphoreGive(xMutexLcd01);
            }
            vTaskDelay(pdMS_TO_TICKS(2000));
            if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
              lcd01.clear();
              lcd01.print("Nhap mat khau:");
              xSemaphoreGive(xMutexLcd01);
            }
          }
          inputPass = "";
        } else if (key == '*') {
          inputPass = "";
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print("Nhap mat khau:");
            xSemaphoreGive(xMutexLcd01);
          }
        } else if (key == 'A') {
          changePassMode = true;
          changeStep = 1;
          inputPass = "";
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print("Nhap MK cu:");
            xSemaphoreGive(xMutexLcd01);
          }
        } else {
          inputPass += key;
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.setCursor(0, 1);
            lcd01.print(inputPass);
            xSemaphoreGive(xMutexLcd01);
          }
        }
      } else {
        // Change password flow
        if (key == '#') {
          if (changeStep == 1) {
            if (inputPass == passwordSet) {
              changeStep = 2;
              inputPass = "";
              if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
                lcd01.clear();
                lcd01.print("Nhap MK moi:");
                xSemaphoreGive(xMutexLcd01);
              }
            } else {
              if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
                lcd01.clear();
                lcd01.print("MK cu sai!");
                xSemaphoreGive(xMutexLcd01);
              }
              buzzerBeep(1, 1000, 0);
              vTaskDelay(pdMS_TO_TICKS(2000));
              if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
                lcd01.clear();
                lcd01.print("Nhap mat khau:");
                xSemaphoreGive(xMutexLcd01);
              }
              changePassMode = false;
              changeStep = 0;
              inputPass = "";
            }
          } else if (changeStep == 2) {
            passwordSet = inputPass;
            sha256Hash(inputPass.c_str(), storedPinHash);
            saveCredentials();
            if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
              lcd01.clear();
              lcd01.print("MK da doi!");
              xSemaphoreGive(xMutexLcd01);
            }
            buzzerBeep(2, 200, 200);
            vTaskDelay(pdMS_TO_TICKS(2000));
            if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
              lcd01.clear();
              lcd01.print("Nhap mat khau:");
              xSemaphoreGive(xMutexLcd01);
            }
            changePassMode = false;
            changeStep = 0;
            inputPass = "";
          }
        } else if (key == '*') {
          inputPass = "";
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print(changeStep == 1 ? "Nhap MK cu:" : "Nhap MK moi:");
            xSemaphoreGive(xMutexLcd01);
          }
        } else {
          inputPass += key;
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.setCursor(0, 1);
            lcd01.print(inputPass);
            xSemaphoreGive(xMutexLcd01);
          }
        }
      }
    }

    vTaskDelay(pdMS_TO_TICKS(50));
  }
}

// ---- Task 4: PCF8574 Buttons (Core 1) ---------------------------------------
void TaskButtons(void* pvParameters) {
  static bool allLightsOn = false;

  for (;;) {
    // Door toggle
    if (pcf.read(BTN_DOOR) == LOW) {
      buzzerBeep(1, 100, 0);
      if (fireDoorOpened) {
        closeDoorSmooth();
        fireDoorOpened = false;
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Cua: DONG");
          xSemaphoreGive(xMutexLcd02);
        }
      } else {
        openDoorSmooth();
        fireDoorOpened = true;
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Cua: MO");
          xSemaphoreGive(xMutexLcd02);
        }
      }
      publishDoorState();
      publishDoorAccessLog("access_granted", "BUTTON", "physical_button");
      failedAttempts = 0;
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // Fan toggle
    if (pcf.read(BTN_FAN) == LOW) {
      buzzerBeep(1, 100, 0);
      fanState = !fanState;
      digitalWrite(RELAY_FAN, fanState ? LOW : HIGH);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print(fanState ? "Quat: ON" : "Quat: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
      publishFanState();
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // Pump toggle
    if (pcf.read(BTN_PUMP) == LOW) {
      buzzerBeep(1, 100, 0);
      pumpState = !pumpState;
      digitalWrite(RELAY_PUMP, pumpState ? LOW : HIGH);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print(pumpState ? "Bom: ON" : "Bom: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
      publishPumpState();
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // All lights toggle
    if (pcf.read(BTN_LIGHT) == LOW) {
      buzzerBeep(1, 100, 0);
      allLightsOn = !allLightsOn;
      if (allLightsOn) {
        lightOutAuto = true;
        lightLiv = true;
        digitalWrite(LIGHT_LIV, HIGH);
        setBedColor(7);
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Tat ca den: ON");
          xSemaphoreGive(xMutexLcd02);
        }
      } else {
        lightOutAuto = false;
        lightLiv = false;
        digitalWrite(LIGHT_LIV, LOW);
        setBedColor(0);
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Tat ca den: OFF");
          xSemaphoreGive(xMutexLcd02);
        }
      }
      publishLightState();
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // Outside light toggle
    if (pcf.read(BTN_L_OUT) == LOW) {
      buzzerBeep(1, 100, 0);
      lightOutAuto = !lightOutAuto;
      if (!lightOutAuto) digitalWrite(LIGHT_OUT, LOW);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print(lightOutAuto ? "Den ngoai: AUTO" : "Den ngoai: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
      publishLightState();
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // Living room light toggle
    if (pcf.read(BTN_L_LIV) == LOW) {
      buzzerBeep(1, 100, 0);
      lightLiv = !lightLiv;
      digitalWrite(LIGHT_LIV, lightLiv ? HIGH : LOW);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        lcd02.print(lightLiv ? "Den khach: ON" : "Den khach: OFF");
        xSemaphoreGive(xMutexLcd02);
      }
      publishLightState();
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // Bedroom color cycle
    if (pcf.read(BTN_L_BED) == LOW) {
      buzzerBeep(1, 100, 0);
      bedColorIndex++;
      if (bedColorIndex > 7) bedColorIndex = 0;
      setBedColor(bedColorIndex);
      if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
        lcd02.clear();
        if (bedColorIndex == 0) lcd02.print("Den ngu: OFF");
        else {
          lcd02.print("Den ngu: Mau ");
          lcd02.print(bedColorIndex);
        }
        xSemaphoreGive(xMutexLcd02);
      }
      publishLightState();
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    // Emergency toggle / Alarm reset
    if (pcf.read(BTN_FIRE) == LOW) {
      buzzerBeep(1, 100, 0);
      if (doorAlarmActive) {
        vTaskDelay(pdMS_TO_TICKS(300));
        resetDoorAlarm();
      } else {
        emergencyMode = !emergencyMode;
        if (emergencyMode) {
          alarmMuted = false;
          digitalWrite(BUZZER_PIN, HIGH);
          digitalWrite(LED_ALARM, HIGH);
          digitalWrite(RELAY_FAN, LOW);
          digitalWrite(RELAY_PUMP, LOW);
          fanState = true;
          pumpState = true;
          if (!fireDoorOpened) {
            openDoorSmooth();
            fireDoorOpened = true;
          }
          lightOutAuto = false;
          lightLiv = false;
          digitalWrite(LIGHT_OUT, LOW);
          digitalWrite(LIGHT_LIV, LOW);
          setBedColor(0);
          if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd02.clear();
            lcd02.print("CANH BAO: ON");
            xSemaphoreGive(xMutexLcd02);
          }
        } else {
          digitalWrite(BUZZER_PIN, LOW);
          digitalWrite(LED_ALARM, LOW);
          digitalWrite(RELAY_FAN, HIGH);
          digitalWrite(RELAY_PUMP, HIGH);
          fanState = false;
          pumpState = false;
          if (fireDoorOpened) {
            closeDoorSmooth();
            fireDoorOpened = false;
          }
          if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd02.clear();
            lcd02.print("CANH BAO: OFF");
            xSemaphoreGive(xMutexLcd02);
          }
          if (xSemaphoreTake(xMutexLcd01, pdMS_TO_TICKS(500)) == pdTRUE) {
            lcd01.clear();
            lcd01.print("Nhap mat khau:");
            xSemaphoreGive(xMutexLcd01);
          }
        }
        publishAllStates();
      }
      vTaskDelay(pdMS_TO_TICKS(300));
    }

    vTaskDelay(pdMS_TO_TICKS(50));
  }
}

// ---- Task 5: Drying Rack (Core 0) -------------------------------------------
void TaskDryingRack(void* pvParameters) {
  for (;;) {
    int rainState = digitalRead(RAIN_SENSOR);
    if (rainState == LOW) {
      if (!dryingOut) {
        Serial.println("🌧 Rain detected -> Retract");
        if (xSemaphoreTake(xMutexServo, pdMS_TO_TICKS(3000)) == pdTRUE) {
          moveServoSmooth(dryServo, 0, 90, 20);
          xSemaphoreGive(xMutexServo);
        }
        dryingOut = true;
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Mua: Thu Gian");
          xSemaphoreGive(xMutexLcd02);
        }
      }
    } else {
      if (dryingOut) {
        Serial.println("☀️ Dry -> Extend");
        if (xSemaphoreTake(xMutexServo, pdMS_TO_TICKS(3000)) == pdTRUE) {
          moveServoSmooth(dryServo, 90, 0, 20);
          xSemaphoreGive(xMutexServo);
        }
        dryingOut = false;
        if (xSemaphoreTake(xMutexLcd02, pdMS_TO_TICKS(500)) == pdTRUE) {
          lcd02.clear();
          lcd02.print("Nang Phoi Do");
          xSemaphoreGive(xMutexLcd02);
        }
      }
    }
    vTaskDelay(pdMS_TO_TICKS(500));
  }
}

// ---- Task 6: Alarm Blink (Core 0) -------------------------------------------
// Maintains LED blink when doorAlarmActive; also beeps periodically.
void TaskAlarm(void* pvParameters) {
  unsigned long lastBeep = 0;
  for (;;) {
    if (doorAlarmActive) {
      digitalWrite(LED_ALARM, (millis() / 500) % 2);
      if (millis() - lastBeep > 2000) {
        lastBeep = millis();
        buzzerBeep(1, 500, 0);
      }
    }
    vTaskDelay(pdMS_TO_TICKS(100));
  }
}

// =============================================================================
//                               S E T U P
// =============================================================================
void setup() {
  Serial.begin(115200);
  Wire.begin(I2C_SDA, I2C_SCL);

  // ---- LCD Init ----
  lcd01.init();
  lcd01.backlight();
  lcd02.init();
  lcd02.backlight();
  lcd01.print("He thong khoi");
  lcd02.print("He thong khoi");
  lcd01.setCursor(0, 1);
  lcd01.print("dong...");
  lcd02.setCursor(0, 1);
  lcd02.print("dong...");
  delay(2000);
  lcd01.clear();
  lcd02.clear();
  lcd01.print("Nhap mat khau:");
  lcd02.print("Active:");

  // ---- Keypad ----
  keypad.begin();

  // ---- RFID ----
  SPI.begin(18, 19, 23, 5);
  mfrc522.PCD_Init();

  // ---- Door Servo ----
  doorServo.setPeriodHertz(50);
  doorServo.attach(SERVO_PIN, 500, 2400);
  doorServo.write(0);

  // ---- Pin Modes ----
  pinMode(FLAME_SENSOR, INPUT);
  pinMode(GAS_SENSOR, INPUT);
  pinMode(RELAY_FAN, OUTPUT);
  pinMode(RELAY_PUMP, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_ALARM, OUTPUT);
  pinMode(LIGHT_LIV, OUTPUT);
  pinMode(LDR_SENSOR, INPUT);
  pinMode(LIGHT_OUT, OUTPUT);
  pinMode(RAIN_SENSOR, INPUT);

  digitalWrite(RELAY_FAN, HIGH);
  digitalWrite(RELAY_PUMP, HIGH);
  digitalWrite(BUZZER_PIN, LOW);
  digitalWrite(LED_ALARM, LOW);
  digitalWrite(LIGHT_OUT, LOW);
  digitalWrite(LIGHT_LIV, LOW);

  // ---- NeoPixel ----
  pixels.begin();
  neoPixelOff();

  // ---- PCF8574 ----
  if (!pcf.begin()) {
    Serial.println("❌ PCF8574 not found!");
    while (1)
      ;
  }
  for (int i = 0; i < 8; i++) pcf.write(i, HIGH);

  // ---- Dry Rack Servo ----
  dryServo.setPeriodHertz(50);
  dryServo.attach(DRY_SERVO_PIN, 500, 2400);
  dryServo.write(0);

  // ---- FreeRTOS Primitives ----
  xMutexLcd01 = xSemaphoreCreateMutex();
  xMutexLcd02 = xSemaphoreCreateMutex();
  xMutexMqtt = xSemaphoreCreateMutex();
  xMutexServo = xSemaphoreCreateMutex();
  xMutexPrefs = xSemaphoreCreateMutex();

  xQueueMqttOut = xQueueCreate(32, sizeof(MqttMsg));
  xQueueMqttIn = xQueueCreate(16, sizeof(MqttCmd));

  // ---- NVS ----
  loadCredentials();

  // ---- WiFi ----
  lcd02.clear();
  lcd02.print("Connecting WiFi");
  WiFi.begin(ssid, pass);
  Serial.print("Connecting to WiFi");
  int retry = 0;
  while (WiFi.status() != WL_CONNECTED && retry < 20) {
    delay(500);
    Serial.print(".");
    retry++;
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n✅ WiFi connected! IP: " + WiFi.localIP().toString());
    lcd02.clear();
    lcd02.print("WiFi: OK");
    wifiConnected = true;
  } else {
    Serial.println("\n❌ WiFi failed!");
    lcd02.clear();
    lcd02.print("WiFi: Fail");
  }

  // ---- MQTT TLS ----
  mqtt.setServer(mqtt_host, mqtt_port);
  mqtt.setCallback(mqttCallback);
  mqtt.setBufferSize(1024);

  delay(1000);

  // ==================== Create FreeRTOS Tasks ====================
  //                    Task             Name          Stack   Param  Prio  Handle  Core
  xTaskCreatePinnedToCore(TaskMQTT, "TaskMQTT", 8192, NULL, 3, NULL, 0);
  xTaskCreatePinnedToCore(TaskSensors, "TaskSensors", 4096, NULL, 2, NULL, 0);
  xTaskCreatePinnedToCore(TaskDryingRack, "TaskDryRack", 2048, NULL, 1, NULL, 0);
  xTaskCreatePinnedToCore(TaskAlarm, "TaskAlarm", 2048, NULL, 2, NULL, 0);
  xTaskCreatePinnedToCore(TaskKeypadRFID, "TaskKeyRFID", 6144, NULL, 3, NULL, 1);
  xTaskCreatePinnedToCore(TaskButtons, "TaskButtons", 4096, NULL, 2, NULL, 1);

  Serial.println("🏠 FreeRTOS Smart Home ready!");
}

// ================== Loop: idle (all logic in tasks) ==================
void loop() {
  vTaskDelay(pdMS_TO_TICKS(10000));
}
